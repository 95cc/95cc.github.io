        "terminal.integrated.fontFamily": "Menlo for Powerline"

    - [rbenv](https://github.com/rbenv/rbenv)
      - `brew install rbenv`
      - 添加配置到`~/.zshrc`中
        
        ```bash
        export PATH="$HOME/.rbenv/bin:$PATH"
        eval "$(rbenv init -)"
        export RUBY_BUILD_MIRROR_URL=https://cache.ruby-china.com
        ```
